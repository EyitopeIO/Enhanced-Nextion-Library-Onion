499 void Nextion::nexLoop(NexTouch *nex_listen_list[]) const
500 {
501  static uint8_t __buffer[10];
502 
503  while (m_nexSerial->available())
504  {
505  __buffer[0] = m_nexSerial->read();
506  m_nexSerial->setTimeout(200);
507  switch(__buffer[0])
508  {
509  case NEX_RET_EVENT_NEXTION_STARTUP:
510  {
511  if(5==m_nexSerial->readBytes(&__buffer[1],5))
512  {
513  if (0x00 == __buffer[1] && 0x00 == __buffer[2] && 0xFF == __buffer[3] && 0xFF == __buffer[4] && 0xFF == __buffer[5])
514  {
515  if(nextionStartupCallback!=nullptr)
516  {
517  nextionStartupCallback();
518  }
519  }
520  }
521  break;
522  }
523  case NEX_RET_EVENT_TOUCH_HEAD:
524  {
525  if(6==m_nexSerial->readBytes(&__buffer[1],6))
526  {
527  if (0xFF == __buffer[4] && 0xFF == __buffer[5] && 0xFF == __buffer[6])
528  {
529  NexTouch::iterate(nex_listen_list, __buffer[1], __buffer[2], __buffer[3]);
530  }
531  }
532  break;
533  }
534  case NEX_RET_CURRENT_PAGE_ID_HEAD:
535  {
536  if(4==m_nexSerial->readBytes(&__buffer[1],4))
537  {
538  if (0xFF == __buffer[2] && 0xFF == __buffer[3] && 0xFF == __buffer[4])
539  {
540  if(currentPageIdCallback!=nullptr)
541  {
542  currentPageIdCallback(__buffer[1]);
543  }
544  }
545  }
546  break;
547  }
548  case NEX_RET_EVENT_POSITION_HEAD:
549  case NEX_RET_EVENT_SLEEP_POSITION_HEAD:
550  {
551  if(8==m_nexSerial->readBytes(&__buffer[1],8))
552  {
553  if (0xFF == __buffer[6] && 0xFF == __buffer[7] && 0xFF == __buffer[8])
